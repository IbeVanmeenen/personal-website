var youtubeVideos, onYouTubeIframeAPIReady, initSlider;

// Youtube
youtubeVideos = {
    done: false,
    initFlag: true,
    init: function() {
        var tag, firstScriptTag;
        // Insert the iframe_api script on first run
        if (youtubeVideos.initFlag) {
            tag = document.createElement('script');
            tag.src = "https://www.youtube.com/iframe_api";
            firstScriptTag = document.getElementsByTagName('script')[0];
            firstScriptTag.parentNode.insertBefore(tag, firstScriptTag);
            youtubeVideos.initFlag = false;
        }
    },
    players: [],
    pauseVideo: function(player) {
        player.pauseVideo();
    }
};

// When the YouTube API is ready to be used
onYouTubeIframeAPIReady = function() {
    // Loop over the Youtube iframes
    $("iframe[src^='http://www.youtube.com/embed/']").each(function(i) {
        var player = new YT.Player($(this).get()[0]);
        $(this).data('index', i);
        youtubeVideos.players.push(player);
    });
};

// Init youtubeVideos
youtubeVideos.init();

// Init Flexslider
initSliders = function() {
    $('#slider').flexslider({
        slideshow: false,
        video: true,
        before: function() {
            // Pause all videos the slider
            $("#slider iframe[src^='http://www.youtube.com/embed/']").each(function(){
                youtubeVideos.pauseVideo(youtubeVideos.players[$(this).data('index')]);
            });
        }
    });
};
