Example 


var aPromiseFunction = function(value) {
    // Store value
    var isOk = value;

    // Make function a promise
    return new Promise(function(resolve, reject) {
        console.log(value);

        if (isOk) {
            // Resolve promise
            resolve();
        } else {
            // Reject promise
            reject();
        }
    });
};

aPromiseFunction(true).then(function() {
    // When resolved
    console.log('awesome');
}, function(error) {
    // When rejected
    console.log(error);
});
